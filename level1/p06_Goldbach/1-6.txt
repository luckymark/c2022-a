#include<stdio.h>
int isPrime(int n)
{
	int i;
	for(i=2;i<n;++i){
	    if(n%i==0)
	        return 0;
    }
    return 1;
}
int isEven(int n)
{
    return n%2==0?1:0;
}
int main()
{
	int n;
	int i;
	int j=0;  
	printf("input an even:");
	scanf("%d",&n);
    while(!isEven(n)){
    	printf("error\n");
    	printf("input an even:");
    	scanf("%d",&n);
	}  
	for(i=2;i<n/2;i++){   
    	if(isPrime(i))
		{
			if (isPrime(n-i))
			{
				printf("哥德巴赫求和：%d=%d+%d\n",n,i,n-i);
				j=1;
			}
		}
	}
	if (j)
		return 0;
	else{
		printf("哥德巴赫猜想不成立");
		return -1;	
	}
}
